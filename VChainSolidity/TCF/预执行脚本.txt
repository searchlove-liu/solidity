        const zeroAddress = '0'.repeat(64);

}

function convertValueToHex(ethValue) {
    try {
        const ethNum = parseFloat(ethValue);
        if (isNaN(ethNum) || ethNum < 0) {
            throw new Error(`无效的ETH值: ${ethValue}`);
        }
        
        // 直接使用输入的value作为转账金额
        const hexValue = BigInt(ethNum).toString(16);

        return "0x" + hexValue;

        return "0x" + "0".repeat(64);

}
